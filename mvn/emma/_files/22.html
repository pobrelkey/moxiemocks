365    @SuppressWarnings("unchecked")
366    public Object[] autoMock(Object... testComponents) {
367        for (Object testInstance : testComponents) {
368            if (valuesOverwrittenByAutoMock.containsKey(testInstance)) {
369                throw new IllegalArgumentException("object has already been autoMock()'ed: " + testInstance);
370            }
371        }
372        ArrayList result = new ArrayList();
373        for (Object testInstance : testComponents) {
374            Map<String, Object> oldValues = new HashMap<String, Object>();
375            valuesOverwrittenByAutoMock.put(testInstance, oldValues);
376            for (Class c = testInstance.getClass(); c != null; c = c.getSuperclass()) {
377                for (Field f : c.getDeclaredFields()) {
378                    Mock mockAnnotation = f.getAnnotation(Mock.class);
379                    Spy spyAnnotation = f.getAnnotation(Spy.class);
380                    AutoMock autoMockAnnotation = f.getAnnotation(AutoMock.class);
381                    boolean isGroup = Group.class.equals(f.getType());
382                    if (mockAnnotation != null || spyAnnotation != null || autoMockAnnotation != null || isGroup) {
383                        if (!f.isAccessible()) {
384                            f.setAccessible(true);
385                        }
386                        try {
387                            Object testObject;
388                            if (autoMockAnnotation != null) {
389                                autoMock(f.get(testInstance));
390                                continue;
391                            } else if (isGroup) {
392                                GroupOptions optionsAnnotation = f.getAnnotation(GroupOptions.class);
393                                testObject = Moxie.group(f.getName(), optionsAnnotation != null ? optionsAnnotation.value() : new MoxieOptions[0]);
394                            } else if (spyAnnotation != null) {
395                                testObject = Moxie.spy(f.get(testInstance), f.getName(), spyAnnotation.value());
396                            } else {
397                                testObject = Moxie.mock(f.getType(), f.getName(), mockAnnotation.value());
398                            }
399                            oldValues.put(f.getName(), f.get(testInstance));
400                            f.set(testInstance, testObject);
401                            result.add(testObject);
402                        } catch (Exception ex) {
403                            throw new MoxieUnexpectedError("Reflection error when auto-mocking field " + f.getName() + " on object " + testInstance, ex);
404                        }
405                    }
406                }
407            }
408        }
409        return result.toArray(new Object[result.size()]);
410    }

414        Collection components;

416            for (Object testInstance : testComponents) {

419                }
420            }

427            Map<String, Object> oldValues = valuesOverwrittenByAutoMock.remove(testInstance);
428            for (Class c = testInstance.getClass(); c != null; c = c.getSuperclass()) {

430                    if (f.getAnnotation(Mock.class) != null || f.getAnnotation(Spy.class) != null || f.getAnnotation(AutoMock.class) != null || Group.class.equals(f.getType())) {

440                        } catch (Exception ex) {
441                            throw new MoxieUnexpectedError("Reflection error when auto-unmocking field " + f.getName() + " on object " + testInstance, ex);
442                        }
443                    }
444                }
445            }
